Preparar la tabla ofertas (productos) para las promociones ALTER TABLE public.ofertas ADD COLUMN IF NOT EXISTS es_promo BOOLEAN DEFAULT false; -- 3. Crear tabla para guardar los clics de WhatsApp (Estadísticas) CREATE TABLE IF NOT EXISTS public.estadisticas_clics ( id UUID DEFAULT extensions.uuid_generate_v4() PRIMARY KEY, negocio_id UUID REFERENCES public.negocios(id) ON DELETE CASCADE, fecha TIMESTAMPTZ DEFAULT NOW() ); -- 4. Dar permisos (RLS) para que la app pueda guardar y contar los clics ALTER TABLE public.estadisticas_clics ENABLE ROW LEVEL SECURITY; CREATE POLICY "Permitir insertar clics" ON public.estadisticas_clics FOR INSERT WITH CHECK (true); CREATE POLICY "Permitir leer clics" ON public.estadisticas_clics FOR SELECT USING (true);
